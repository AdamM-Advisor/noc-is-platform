## 1. DATABASE SCHEMA — Tabel Baru

### 1A. `ext_weather` (Data Cuaca BMKG)

    CREATE TABLE IF NOT EXISTS ext_weather (
        id INTEGER PRIMARY KEY,
        date DATE NOT NULL,
        province VARCHAR NOT NULL,          -- "Jawa Barat", "DKI Jakarta"
        city VARCHAR,                       -- "Bandung", "Jakarta Selatan"
        -- Metrik cuaca harian
        rainfall_mm DOUBLE,                 -- curah hujan (mm)
        temperature_avg_c DOUBLE,           -- suhu rata-rata (°C)
        temperature_max_c DOUBLE,
        temperature_min_c DOUBLE,
        humidity_avg_pct DOUBLE,            -- kelembapan (%)
        wind_speed_avg_kmh DOUBLE,          -- kecepatan angin
        weather_condition VARCHAR,          -- "Cerah"/"Berawan"/"Hujan"/"Hujan Lebat"/"Badai"
        -- Derived
        is_extreme BOOLEAN DEFAULT FALSE,   -- curah hujan > 50mm ATAU angin > 40km/h
        -- Metadata
        source VARCHAR DEFAULT 'BMKG',
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    );
    
    CREATE INDEX IF NOT EXISTS idx_weather_date ON ext_weather(date);
    CREATE INDEX IF NOT EXISTS idx_weather_province ON ext_weather(province);

### 1B. `ext_pln_outage` (Data Gangguan PLN)

    CREATE TABLE IF NOT EXISTS ext_pln_outage (
        id INTEGER PRIMARY KEY,
        date DATE NOT NULL,
        province VARCHAR NOT NULL,
        city VARCHAR,
        district VARCHAR,
        -- Detail gangguan
        outage_type VARCHAR,                -- "Pemadaman Bergilir"/"Gangguan"/"Pemeliharaan"
        start_time TIMESTAMP,
        end_time TIMESTAMP,
        duration_hours DOUBLE,
        affected_area VARCHAR,              -- deskripsi area terdampak
        -- Metadata
        source VARCHAR DEFAULT 'PLN',
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    );
    
    CREATE INDEX IF NOT EXISTS idx_pln_date ON ext_pln_outage(date);
    CREATE INDEX IF NOT EXISTS idx_pln_province ON ext_pln_outage(province);

### 1C. `ext_calendar` (Kalender & Hari Libur)

    CREATE TABLE IF NOT EXISTS ext_calendar (
        date DATE PRIMARY KEY,
        year INTEGER NOT NULL,
        month INTEGER NOT NULL,
        day INTEGER NOT NULL,
        day_of_week INTEGER NOT NULL,       -- 0=Sun, 6=Sat
        week_of_year INTEGER,
        -- Klasifikasi
        is_weekend BOOLEAN DEFAULT FALSE,
        is_holiday BOOLEAN DEFAULT FALSE,
        is_cuti_bersama BOOLEAN DEFAULT FALSE,
        is_ramadan BOOLEAN DEFAULT FALSE,
        holiday_name VARCHAR,               -- "Idul Fitri 1446 H", "Natal"
        day_type VARCHAR NOT NULL,          -- "Kerja"/"Akhir Pekan"/"Libur Nasional"/"Cuti Bersama"
        -- Metadata
        source VARCHAR DEFAULT 'built_in',
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    );

### 1D. `ext_annotations` (Anotasi pada Trend)

    CREATE TABLE IF NOT EXISTS ext_annotations (
        id INTEGER PRIMARY KEY,
        -- Waktu
        date DATE NOT NULL,
        date_end DATE,                      -- NULL jika event 1 hari
        -- Scope (opsional — NULL berarti berlaku nasional)
        area_id VARCHAR,
        regional_id VARCHAR,
        province VARCHAR,
        -- Konten
        annotation_type VARCHAR NOT NULL,   -- "weather"/"pln"/"holiday"/"custom"/"incident"
        title VARCHAR NOT NULL,             -- "Banjir Jakarta", "Lebaran 2025"
        description VARCHAR,
        severity VARCHAR DEFAULT 'info',    -- "info"/"warning"/"critical"
        -- Display
        color VARCHAR,                      -- hex color untuk marker
        icon VARCHAR,                       -- emoji: 🌧️ ⚡ 🕌 📌
        show_on_chart BOOLEAN DEFAULT TRUE,
        -- Metadata
        source VARCHAR DEFAULT 'manual',    -- "manual"/"auto_weather"/"auto_pln"/"built_in"
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    );
    
    CREATE INDEX IF NOT EXISTS idx_annotations_date ON ext_annotations(date);

## 3. UPLOAD DATA CUACA

### 3A. Upload Flow

    POST /api/external/weather/upload
      ← File CSV/Excel dengan kolom:
        date, province, city, rainfall_mm, temperature_avg_c,
        humidity_avg_pct, wind_speed_avg_kmh, weather_condition
    
      → Validasi:
        • date format valid
        • province ada di Indonesia (fuzzy match OK)
        • angka valid (rainfall ≥ 0, temperature reasonable)
    
      → Auto-derive:
        • is_extreme = rainfall_mm > 50 OR wind_speed_avg_kmh > 40
    
      → Auto-generate annotations untuk extreme weather:
        INSERT INTO ext_annotations (...) 
        SELECT date, province, 'weather', 
               'Cuaca Ekstrem: ' || weather_condition,
               'Curah hujan ' || rainfall_mm || 'mm',
               'warning', '#FF6B00', '🌧️', TRUE, 'auto_weather'
        FROM ext_weather WHERE is_extreme = TRUE
    
      → Return: {imported: 365, extreme_days: 23, annotations_created: 23}

### 3B. Template & Format

    GET /api/external/weather/template
      → Download CSV template:
    
      date,province,city,rainfall_mm,temperature_avg_c,humidity_avg_pct,wind_speed_avg_kmh,weather_condition
      2025-01-01,Jawa Barat,Bandung,15.2,24.5,78,12,Berawan
      2025-01-02,Jawa Barat,Bandung,45.8,23.1,92,25,Hujan Lebat
      2025-01-01,DKI Jakarta,,22.3,28.2,75,8,Hujan

## 4. UPLOAD DATA PLN

### 4A. Upload Flow

    POST /api/external/pln/upload
      ← File CSV/Excel dengan kolom:
        date, province, city, district, outage_type,
        start_time, end_time, duration_hours, affected_area
    
      → Auto-generate annotations untuk gangguan PLN:
        INSERT INTO ext_annotations (...)
        SELECT date, province, 'pln',
               outage_type || ' PLN: ' || COALESCE(city, province),
               'Durasi: ' || duration_hours || ' jam. ' || affected_area,
               CASE WHEN duration_hours > 6 THEN 'critical'
                    WHEN duration_hours > 2 THEN 'warning'
                    ELSE 'info' END,
               '#FFD700', '⚡', TRUE, 'auto_pln'
        FROM ext_pln_outage
    
      → Return: {imported: 120, annotations_created: 120}

## 5. ANNOTATION SYSTEM

### 5A. Manual Annotation

    POST /api/external/annotation
      ← Body: {
          date: "2025-03-15",
          date_end: null,            // null = 1 hari
          area_id: "AREA2",         // null = nasional
          annotation_type: "custom",
          title: "Cutover Perangkat R03",
          description: "Penggantian perangkat utama di 50 site R03",
          severity: "warning",
          icon: "📌"
        }
      → Return: created annotation
    
    GET /api/external/annotations
      ← Query: ?from=2025-01-01 &to=2025-07-31 &type=weather,pln,holiday
      → Return: list annotations dalam range, sorted by date
    
    PUT /api/external/annotation/{id}
      → Update annotation
    
    DELETE /api/external/annotation/{id}
      → Delete annotation

### 5B. Auto-Generated Annotations

Sistem otomatis membuat annotations dari:

    1. HARI LIBUR (dari ext_calendar):
       → Setiap hari libur nasional → annotation type="holiday"
       → icon: 🕌 (Islam), ⛪ (Kristen), 🛕 (Hindu/Buddha), 🏛️ (Nasional)
       → severity: "info"
       → Auto-generate saat calendar di-seed
    
    2. CUACA EKSTREM (dari ext_weather):
       → rainfall > 50mm OR wind > 40km/h
       → type="weather", severity="warning", icon="🌧️"
       → Auto-generate saat weather data di-upload
    
    3. GANGGUAN PLN (dari ext_pln_outage):
       → Setiap record PLN
       → type="pln", severity berdasarkan durasi, icon="⚡"
       → Auto-generate saat PLN data di-upload
    
    4. PERIODE RAMADAN:
       → Date range Ramadan → annotation type="holiday"
       → title="Bulan Ramadan", icon="🕌"
       → Penting karena pola gangguan berubah saat Ramadan

### 5C. Annotation Display on Charts

Annotations ditampilkan sebagai **markers pada trend line charts** (di Prompt 7 nanti):

    Cara kerja:
    1. Frontend request: GET /api/external/annotations?from=&to=
    2. Untuk setiap annotation, tampilkan:
       - Vertical line/band pada tanggal di chart
       - Icon di atas chart
       - Tooltip saat hover: title + description
    
    Contoh pada trend line SLA bulanan:
    
      SLA%
      95 ┤                    
      93 ┤──●──●           🕌📌
      91 ┤        ──●──●──│──●──●──
      89 ┤                │        ──●
      87 ┤                │
         ├──┬──┬──┬──┬──┬──┬──┬──┬──
          Jan Feb Mar Apr Mei Jun Jul Aug Sep
                          │
                          └── "Idul Fitri + Cutover R03"
                              (vertikal band kuning)
    
    Visualisasi:
      • info    → garis abu tipis dotted
      • warning → garis kuning/oranye solid  
      • critical→ band merah semi-transparan
      • Holiday → garis biru dotted + icon

## 6. KORELASI SEDERHANA

### 6A. Cuaca × Gangguan

    GET /api/external/correlation/weather
      ← Query: ?province=Jawa Barat &from=2025-01-01 &to=2025-06-30
    
      → Hitung:
        1. Join ext_weather (per date, per province) dengan
           noc_tickets (aggregated per date, per area/regional)
        2. Mapping province → area_id (via master_site lokasi)
           ATAU simplified: province → regional manual mapping
        3. Hitung korelasi:
           - Pearson correlation: rainfall_mm vs daily_ticket_count
           - Pearson correlation: is_extreme vs daily_ticket_count
        4. Buat scatter plot data points
    
      → Return: {
          correlation_rainfall_vs_tickets: 0.42,  // 0-1, makin tinggi = makin korelasi
          correlation_extreme_vs_tickets: 0.68,
          interpretation: "Korelasi moderat antara curah hujan dan jumlah gangguan",
          data_points: [
            {date: "2025-01-01", rainfall_mm: 15.2, tickets: 850},
            ...
          ],
          extreme_impact: {
            avg_tickets_normal: 800,
            avg_tickets_extreme: 1250,
            increase_pct: 56.3
          }
        }

### 6B. PLN × Gangguan

    GET /api/external/correlation/pln
      ← Query: ?province= &from= &to=
    
      → Hitung:
        1. Join ext_pln_outage dengan noc_tickets (per date, area)
        2. Bandingkan ticket volume pada hari PLN gangguan vs normal
        3. Filter rc_category = "Power" untuk korelasi lebih spesifik
    
      → Return: {
          correlation_pln_vs_tickets: 0.71,
          correlation_pln_vs_power_rc: 0.85,
          interpretation: "Korelasi tinggi antara gangguan PLN dan tiket RC Power",
          pln_impact: {
            avg_tickets_no_pln: 800,
            avg_tickets_with_pln: 1450,
            avg_power_rc_no_pln: 150,
            avg_power_rc_with_pln: 520,
            increase_pct: 81.3
          }
        }

### 6C. Hari Kerja vs Libur

    GET /api/external/correlation/calendar
      ← Query: ?from= &to=
    
      → Hitung:
        1. Join ext_calendar dengan noc_tickets (per date)
        2. Bandingkan:
           - Hari Kerja vs Akhir Pekan vs Libur Nasional
           - Ramadan vs Non-Ramadan
    
      → Return: {
          avg_tickets_weekday: 850,
          avg_tickets_weekend: 620,
          avg_tickets_holiday: 580,
          avg_tickets_ramadan: 920,     // bisa naik karena pola konsumsi listrik
          weekend_reduction_pct: -27.1,
          holiday_reduction_pct: -31.8,
          ramadan_increase_pct: 8.2,
          interpretation: "Volume turun ~30% saat weekend/libur, naik ~8% saat Ramadan"
        }

## 8. API ENDPOINTS SUMMARY

    # Calendar
    POST   /api/external/calendar/generate     ← {year}
    GET    /api/external/calendar              ← ?year=&month=&type=
    PUT    /api/external/calendar/holiday      ← update/add holiday
    DELETE /api/external/calendar/holiday/{date}
    
    # Weather
    POST   /api/external/weather/upload        ← CSV/Excel file
    GET    /api/external/weather               ← ?from=&to=&province=
    GET    /api/external/weather/template      → download template CSV
    DELETE /api/external/weather               ← ?from=&to=&province= (clear data)
    
    # PLN
    POST   /api/external/pln/upload            ← CSV/Excel file
    GET    /api/external/pln                   ← ?from=&to=&province=
    GET    /api/external/pln/template          → download template CSV
    DELETE /api/external/pln                   ← ?from=&to= (clear data)
    
    # Annotations
    GET    /api/external/annotations           ← ?from=&to=&type=&area_id=
    POST   /api/external/annotation            ← create manual annotation
    PUT    /api/external/annotation/{id}       ← update
    DELETE /api/external/annotation/{id}       ← delete
    
    # Correlation
    GET    /api/external/correlation/weather   ← ?province=&from=&to=
    GET    /api/external/correlation/pln       ← ?province=&from=&to=
    GET    /api/external/correlation/calendar  ← ?from=&to=

## CATATAN PENTING

1. **OPSIONAL**: Semua fitur di prompt ini bersifat opsional. Platformberfungsi penuh tanpa data cuaca/PLN. Jika ext_weather kosong,tab Korelasi Cuaca menampilkan: "Upload data cuaca untuk analisis."
  
2. **Mapping Provinsi → Area**: Korelasi butuh mapping lokasi.
  
  * Jika master_site punya koordinat → auto-map via provinsi
  * Jika tidak → user bisa set manual mapping provinsi → area/regional
  * Simplified: hardcode mapping provinsi → area_id utama
3. **Annotations di chart**: Implementasi visual anotasi pada chartakan dilakukan di **Prompt 7** (Profiler Temporal). Di prompt inikita hanya menyiapkan data + API. Frontend Prompt 7 akan consume`GET /api/external/annotations?from=&to=` dan render marker.
  
4. **Korelasi = analisis sederhana**: Pearson correlation + rata-rata comparison.Bukan advanced statistics. Cukup untuk indikasi awal.
  
5. **Calendar generate**: saat schema init → otomatis generate 2025 + 2026.Hari libur di-seed. User bisa add/edit/delete holidays.
  
6. **Performance**: ext_weather + ext_pln_outage relatif kecil(~365-730 baris/tahun). Tidak ada concern performa.
