```python
def generate_summary_narrative(entity, kpis, behavior, children):
    """Template-based narasi. BUKAN AI — konsisten dan auditable."""
    parts = []
    
    # Part 1: Volume & SLA
    parts.append(
        f"{entity.name} memproses rata-rata {kpis.avg_volume:,.0f} "
        f"tiket/{kpis.granularity_label} dengan SLA {kpis.sla_pct:.1f}%"
    )
    delta = kpis.sla_pct - kpis.sla_target
    if delta >= 0:
        parts.append(f"({delta:.1f}pp di atas target {kpis.sla_target}%)")
    else:
        parts.append(f"({abs(delta):.1f}pp di bawah target {kpis.sla_target}%)")
    
    # Part 2: Trend
    if kpis.sla_trend_slope is not None and abs(kpis.sla_trend_slope) > 0.5:
        direction = "menurun" if kpis.sla_trend_slope < 0 else "membaik"
        parts.append(
            f"Tren SLA {direction} {abs(kpis.sla_trend_slope):.1f}pp/bulan "
            f"selama {kpis.sla_trend_months} bulan terakhir."
        )
    
    # Part 3: Problem child
    if children and children.worst:
        w = children.worst
        parts.append(
            f"{w.count} dari {children.total} {children.type_label} "
            f"({w.name}) berstatus Chronic dan menyumbang "
            f"{w.contribution_pct:.0f}% total tiket."
        )
    
    return " ".join(parts)
```

## 5. BEHAVIOR LABELING ENGINE (`backend/services/behavior_service.py`)

### 5A. 6 Labels

```
🔴 CHRONIC       — Gangguan kronis berulang (volume tinggi 3+ bulan, SLA < 90%)
📉 DETERIORATING — Tren memburuk (SLA slope < -0.5pp/bulan selama 2+ bulan)
🟠 SPORADIC      — Volume rendah tapi severity tinggi (>30% Critical/Major)
🟡 SEASONAL      — Pola musiman terdeteksi (CV<0.5, peak factor>1.3)
📈 IMPROVING     — Tren membaik (SLA slope > +0.5pp/bulan selama 2+ bulan)
🟢 HEALTHY       — Kondisi sehat (SLA ≥ 95% atau SLA ≥ target, volume terkendali)
```

### 5B. Classification Rules

```python
def classify_behavior(entity_level, months_data):
    """
    Evaluasi berurutan — pertama yang match menang.
    Minimal 2 bulan data untuk klasifikasi.
    """
    avg_tickets = mean([m['total_tickets'] for m in months_data])
    avg_sla = mean([m['sla_pct'] for m in months_data])
    sla_slope = linear_slope([m['sla_pct'] for m in months_data])
    
    # Volume threshold per level:
    # site=20, to=500, nop=2000, regional=10000, area=30000
    high_thr = get_volume_threshold(entity_level)
    
    # 1. CHRONIC: tinggi 3+ bulan konsekutif + SLA < 90%
    consecutive_high = count_consecutive_above(
        [m['total_tickets'] for m in months_data], high_thr)
    if consecutive_high >= 3 and avg_sla < 90:
        return {"label": "CHRONIC", "reason": f"Volume >{high_thr}/bln, {consecutive_high} bln, SLA {avg_sla:.1f}%"}
    
    # 2. DETERIORATING: SLA slope < -0.5pp/bulan, 2+ bulan
    if sla_slope < -1.5 and len(months_data) >= 3:
        return {"label": "DETERIORATING", "reason": f"SLA turun {abs(sla_slope):.1f}pp/bln, {len(months_data)} bulan"}
    if sla_slope < -0.5 and len(months_data) >= 2:
        return {"label": "DETERIORATING", "reason": f"SLA turun {abs(sla_slope):.1f}pp/bln"}
    
    # 3. SPORADIC: volume rendah tapi >30% Critical+Major
    crit_maj_pct = calc_critical_major_pct(months_data)
    if avg_tickets <= high_thr * 0.3 and crit_maj_pct > 30:
        return {"label": "SPORADIC", "reason": f"Volume rendah, {crit_maj_pct:.0f}% severity tinggi"}
    
    # 4. SEASONAL: CV < 0.5 dan ada peak > 1.3x average
    cv = coefficient_of_variation([m['total_tickets'] for m in months_data])
    if cv < 0.5 and len(months_data) >= 4:
        peak_factor = max([m['total_tickets'] for m in months_data]) / max(avg_tickets, 1)
        if peak_factor > 1.3:
            return {"label": "SEASONAL", "reason": f"Peak factor {peak_factor:.1f}×"}
    
    # 5. IMPROVING: SLA slope > +0.5pp/bulan
    if sla_slope > 0.5 and len(months_data) >= 2:
        return {"label": "IMPROVING", "reason": f"SLA naik {sla_slope:.1f}pp/bln"}
    
    # 6. HEALTHY
    if avg_sla >= 95 or (avg_sla >= 90 and avg_tickets <= high_thr * 0.5):
        return {"label": "HEALTHY", "reason": f"SLA {avg_sla:.1f}%"}
    
    # FALLBACK
    if avg_sla >= 90:
        return {"label": "HEALTHY", "reason": f"SLA memenuhi target"}
    else:
        return {"label": "DETERIORATING", "reason": f"SLA {avg_sla:.1f}% di bawah target"}
```

## 6. KPI INTERPRETATION RULES

### 6A. SLA Compliance

```
≥ target         → 🟢 good     "Memenuhi target"
≥ target-2pp     → 🟡 warning  "Mendekati batas, selisih {delta}pp"
≥ target-5pp     → 🟠 alert    "Di bawah target {delta}pp, perlu tindakan"
< target-5pp     → 🔴 critical "Kritis, jauh di bawah target {delta}pp"
```

### 6B. MTTR (minutes)

```
≤ 240  (4j)      → 🟢 good     "Baik"
≤ 720  (12j)     → 🟡 warning  "Perlu perhatian"
  + decomposition: "Bottleneck: {longest_comp} ({val}min, {pct}%)"
≤ 1440 (24j)     → 🟠 alert    "Lambat"
> 1440            → 🔴 critical "Sangat lambat, investigasi proses perbaikan"
```

### 6C. Escalation Rate

```
≤ 3%             → 🟢 good     "Terkendali"
≤ 7%             → 🟡 warning  "Di atas normal (3%)"
> 7%             → 🔴 critical "Tinggi, review kompetensi/kompleksitas"
```

### 6D. Auto Resolve Rate

```
≥ 60%            → 🟢 good     "Otomasi efektif"
≥ 40%            → 🟡 warning  "Ruang peningkatan otomasi"
< 40%            → 🟠 alert    "Rendah, banyak manual"
```

### 6E. Repeat Incident

```
≤ 10%            → 🟢 good     "Minimal"
≤ 25%            → 🟡 warning  "RC perlu ditinjau"
> 25%            → 🔴 critical "Tinggi — root cause belum terselesaikan"
```

### 6F. Volume (kontekstual, MoM change)

```
change > +10%    → 🟡 warning  "Naik signifikan, periksa insiden massal"
|change| ≤ 10%   → ⚪ neutral  "Stabil"
change < -10%    → 🟢 good     "Menurun"
```

## 9. RECOMMENDATION ENGINE

```python
def generate_recommendations(kpis, behavior, peer, children) -> list:
    """Max 5 rekomendasi, sorted by priority (critical first)."""
    recs = []
    
    # SLA gap
    if kpis.sla_pct < kpis.sla_target:
        delta = kpis.sla_target - kpis.sla_pct
        recs.append({"priority": "critical" if delta > 5 else "warning",
            "icon": "🔴" if delta > 5 else "🟡",
            "text": f"SLA gap {delta:.1f}pp — fokus perbaikan pada {children.worst.name if children else 'entitas terbawah'}."})
    
    # MTTR bottleneck
    if kpis.avg_mttr_min > 720:
        longest = max(kpis.mttr_decomposition.items(), key=lambda x: x[1])
        recs.append({"priority": "warning", "icon": "🟡",
            "text": f"MTTR {kpis.avg_mttr_min:.0f}m — bottleneck: {longest[0]} ({longest[1]:.0f}m, {longest[1]/kpis.avg_mttr_min*100:.0f}%)."})
    
    # Deteriorating children
    if children:
        n_det = sum(1 for c in children.list if c.behavior == "DETERIORATING")
        if n_det > 0:
            recs.append({"priority": "warning", "icon": "🟡",
                "text": f"{n_det} {children.type_label} memburuk — monitor dan intervensi."})
    
    # High escalation
    if kpis.escalation_pct > 7:
        recs.append({"priority": "critical", "icon": "🔴",
            "text": f"Eskalasi {kpis.escalation_pct:.1f}% (>7%) — review kompetensi/kompleksitas."})
    
    # High repeat
    if kpis.repeat_pct > 25:
        recs.append({"priority": "critical", "icon": "🔴",
            "text": f"Repeat {kpis.repeat_pct:.1f}% — RC belum solved, RCA mendalam diperlukan."})
    
    recs.sort(key=lambda r: {"critical": 0, "warning": 1, "info": 2}.get(r['priority'], 9))
    return recs[:5]
```

## 12. REPEAT INCIDENT DETECTION

```sql
-- Repeat = site dengan >1 tiket dalam 7 hari untuk fault level yang sama
WITH ticket_gaps AS (
  SELECT site_id, fault_level, occured_time,
    LAG(occured_time) OVER (
      PARTITION BY site_id, fault_level ORDER BY occured_time
    ) as prev_time
  FROM noc_tickets
  WHERE calc_regional_id = 'R03'
    AND calc_year_month BETWEEN '2025-01' AND '2025-07'
)
SELECT COUNT(*) as repeat_count
FROM ticket_gaps
WHERE EXTRACT(EPOCH FROM (occured_time - prev_time)) / 3600 < 168;
-- 168 hours = 7 days

-- repeat_pct = repeat_count / total_tickets * 100
```
